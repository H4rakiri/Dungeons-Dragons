// Функция для получения случайного элемента из массива
function getRandomElement(array) {
    return array[Math.floor(Math.random() * array.length)];
}

// Функция генерации дома
function generateHouse() {
    const owner = getRandomElement(ownerTypes);
    return {
        type: getRandomElement(houseTypes),
        condition: getRandomElement(houseConditions),
        owner: owner,
        contents: getRandomElement(houseContents),
        features: getRandomElement(houseFeatures),
        secrets: getRandomElement(houseSecrets),
        events: getRandomElement(houseEvents)
    };
}

// Функция отображения дома
function displayHouse(house) {
    // Очищаем предыдущее описание
    const container = dv.container;
    container.innerHTML = '';

    // Создаем кнопку

    button.style.padding = '10px 20px';
    button.style.fontSize = '16px';
    button.style.margin = '10px 0';
    button.style.cursor = 'pointer';
    button.style.backgroundColor = '#4CAF50';
    button.style.color = 'white';
    button.style.border = 'none';
    button.style.borderRadius = '4px';
    button.onclick = () => {
        const newHouse = generateHouse();
        displayHouse(newHouse);
    };

    // Создаем новое описание
    dv.paragraph(`
## Описание деревенского дома

Это ${house.type}. Дом ${house.condition}.

### Хозяин

Внимательно осмотревшись, вы замечаете множество интересных деталей: на стенах висят старые фотографии в деревянных рамках,
на полках расставлены глиняные горшки с засушенными травами, а на подоконниках стоят горшки с цветами.
В воздухе приятно пахнет свежеиспеченным хлебом и сушеными травами.
