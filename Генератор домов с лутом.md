function getRandomElement(array) {

    return array[Math.floor(Math.random() * array.length)];

}

function generateHouse() {

    const owner = getRandomElement(ownerTypes);

    // Определяем массив лута по профессии

    return {

        type: getRandomElement(houseTypes),

        owner: owner,

        contents: getRandomElement(houseContents),

## Как использовать

  

1. Откройте этот файл в Obsidian

2. Убедитесь, что у вас установлен плагин Dataview

3. Нажмите кнопку "Сгенерировать новый дом" для получения нового описания

4. Каждый новый дом будет заменять предыдущее описание

  

## О генераторе

  

Генератор создаёт подробные и разнообразные описания обычных деревенских домов, учитывая:

  

- Тип и внешний вид дома

- Состояние постройки

- Профессию и характер владельца

- Внутреннее убранство и особенности

- Интересные детали и текущие события

- Тематический лут, который можно найти в доме

  

Все характеристики подобраны так, чтобы создавать атмосферные описания деревенских домов для ваших приключений.
