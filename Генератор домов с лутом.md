// Функция для получения случайных уникальных элементов из массива
function getRandomUniqueElements(array, count) {
    const arr = [...array];
    const result = [];
    for (let i = 0; i < count && arr.length > 0; i++) {
        const idx = Math.floor(Math.random() * arr.length);
        result.push(arr.splice(idx, 1)[0]);
    }
    return result;
}

// Функция генерации дома
function generateHouse() {
    const owner = getRandomElement(ownerTypes);
    let lootArray = lootByProfession[owner.profession] || [];
    // Добавим шанс на универсальный или ценный лут
    let lootPool = [...lootArray];
    if (Math.random() < 0.7) {
        lootPool = lootPool.concat(genericLoot);
    }
    // Сколько предметов лута генерировать (от 5 до 10)
    const lootCount = 5 + Math.floor(Math.random() * 6);
    const loot = getRandomUniqueElements(lootPool, lootCount);
    return {
        type: getRandomElement(houseTypes),
        condition: getRandomElement(houseConditions),
        owner: owner,
        contents: getRandomElement(houseContents),
        features: getRandomElement(houseFeatures),
        secrets: getRandomElement(houseSecrets),
        events: getRandomElement(houseEvents),
        wall: getRandomElement(wallDecor),
        shelf: getRandomElement(shelfItems),
        windowsill: getRandomElement(windowsillItems),
        smell: getRandomElement(smells),
        floor: getRandomElement(floorItems),
        table: getRandomElement(tableItems),
        corner: getRandomElement(cornerItems),
        loot: loot
    };
}

// Функция отображения дома
function displayHouse(house) {
    // Очищаем предыдущее описание
    const container = dv.container;
    container.innerHTML = '';

    // Создаем кнопку
    const button = dv.el('button', 'Сгенерировать новый дом');

    button.style.fontSize = '16px';
    button.style.margin = '10px 0';
    button.style.cursor = 'pointer';
    button.style.backgroundColor = '#4CAF50';
    button.style.color = 'white';
    button.style.border = 'none';

    button.onclick = () => {

3. Нажмите кнопку "Сгенерировать новый дом" для получения нового описания
4. Каждый новый дом будет заменять предыдущее описание

## О генераторе

Генератор создаёт подробные и разнообразные описания обычных деревенских домов, учитывая:

- Тип и внешний вид дома
- Состояние постройки
- Профессию и характер владельца
- Внутреннее убранство и особенности
- Интересные детали и текущие события
- Тематический лут, который можно найти в доме

Все характеристики подобраны так, чтобы создавать атмосферные описания деревенских домов для ваших приключений.
